prog -> stmt. 
stmt -> IF expr THEN stmt ELSE stmt
            | WHILE expr DO stmt
            | INPUT ID
            | ID ASSIGN expr
            | WRITE expr
            | BEGIN stmtlist END. 
stmtlist -> stmtlist_.
stmtlist_ -> stmt SEMICOLON stmtlist_
            |. 
expr -> term expr_. 
expr_ -> ADD term expr_
            | SUB term expr_
            |.
term -> factor term_. 
term_ -> MUL factor term_
            | DIV factor term_
            |.
factor -> LPAR expr RPAR
            | ID
            | NUM
            | SUB NUM.
